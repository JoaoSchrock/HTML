<!-- 
Bloco de código do botão voltar ao topo, com animação de scrolling.
Testado como widget HTML/Javascript no Blogger e no Tumblr (modelos padrão).
Browsers IE9, Chrome 21.0.1180.83 e Firefox.

1-
No layout do seu Blogger, adicione um widget HTML/Javascript no fim da página, e cole o código abaixo na caixa "Conteúdo". Não precisa informar o Título.
No Tumblr, clique em "Customize" e depois em "Edit HTML". No fim do código, cole o código abaixo antes de "</body>".

2-
Substitua <<URL of your arrow image>> pela URL da sua imagem.
Por exemplo: http://static.tumblr.com/44uea6e/DhOm9887f/backtop.png

3- 
Salve o seu layout no blogger ou o seu modelo no tumblr, e recarregue a página do seu blog.
O botão vai aparecer depois de rolar um pouco. Quando clicar, a tela vai deslizar de volta ao topo.

Exemplo de blogs com o código funcionando: 
http://www.animeblog.com.br/
http://animeblog-br.tumblr.com/

Observação:
Lembre-se, se você já usa jQuery no seu blog, vai precisar testar e adaptar com a sua versão.
Essa funcionalidade foi testada com o jQuery 1.6.2.
-->

<!-- Code block for Back to top -->
<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.6.2/jquery.min.js"></script>

<style type="text/css">
#backtopo {
    display: scroll;
    position: fixed;
    top: 30px;
    right: 10px;
}
</style>

<div id="backtopo">
<a href="#" title="Back to Top"><img src="<<URL of your arrow image>>" /></a>
</div>

<script>
$(document).ready(function(){
    
    // hide #back-top first
    $('#backtopo').hide();
       
    // fade in #back-top
    $(function () {
        $(window).scroll(function () {
            if ($(this).scrollTop() > 150) {
                $('#backtopo').fadeIn();
            } else {
                $('#backtopo').fadeOut();
            }
        });

        // scroll body to 0px on click
        $('#backtopo').click(function () {
            $('body,html').animate({
                scrollTop: 0
            }, 800);
            return false;
        });
    });

});
</script>
